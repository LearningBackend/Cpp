Синтаксис Switch:

switch(вираження)
{
case значення1:
     дія1;
     break;

case значення2:
     дія2;
     break;

case значення3:
     дія3;
     break;
default:
    дія_за_замовчуванням
    break;
}

switch (a)
	{
	case 1:
	{
		cout << "Ви ввели 1" << endl;
	}// можна в дужках записувати, щоб було зручно читати

	case 2:
		cout << "Ви ввели 2" << endl;
		break;
     

	default:
